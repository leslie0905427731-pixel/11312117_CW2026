# 题目
a004. 文文的求婚
# 解題思路 
使用 while 循环来进行反复输入直到 EOF
# 解題編號
18141647
# 程式碼
```cpp=
#include <iostream>
int main() {
    int i;
    while (std::cin >> i)
    {
        if ((i % 4 == 0 && i % 100 != 0) || (i % 400 == 0)) std::cout << "閏年" << std::endl;
        else std::cout << "平年" << std::endl;
    }
    
    return 0;
}
```
